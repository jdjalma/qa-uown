```typescript
/** Get merchant row by ref code (read-only). Used for baseline + post-save verification. */
async getMerchantByRefCode(refMerchantCode: string): Promise<{
  pk: number;
  ref_merchant_code: string;
  inventory_category: string | null;
  is_active: boolean;
  created_at: Date;
  created_by: string | null;
  updated_at: Date | null;
  updated_by: string | null;
  cloned_from: number | null;
} | null> {
  return this.queryOne(
    `SELECT pk, ref_merchant_code, inventory_category, is_active,
            created_at, created_by, updated_at, updated_by, cloned_from
     FROM uown_merchant
     WHERE ref_merchant_code = $1`,
    [refMerchantCode],
  );
}

/** Count merchants matching a ref code вЂ” used to assert absence (count=0). */
async countMerchantByRefCode(refMerchantCode: string): Promise<number> {
  return this.getSingleNumber(
    `SELECT COUNT(*) FROM uown_merchant WHERE ref_merchant_code = $1`,
    [refMerchantCode],
  );
}
```

## Edge Cases / Notes
- **Toast race:** save-click в†’ toast may appear before `waitForRequest` resolves. Always arm request/response waiters BEFORE clicking SAVE.
- **React-select portals:** options render in a portal outside the form DOM вЂ” use `SELECTORS.filterMenuPortal` parent scope (or `document.body` scoped locator).
- **Case normalization:** backend uppercases `inventoryCategory` on persist. Assert with `.toUpperCase()` on both sides or use case-insensitive matcher.
- **`OW90218-0001` state:** if another test is mid-flight against this merchant, CT-06 baseline may be transiently inconsistent вЂ” enforced via `describe.serial`.
- **Worker cleanup:** CT-03 intentionally does NOT delete the persisted merchant вЂ” unique `refMerchantCode` (including `RUN_ID` + timestamp) prevents collisions across reruns. Data is not considered garbage since it is flagged in name (`RU1262_*`).
- **CT-07 cleanup idempotency:** if step 12 fails mid-revert, next run's CT-07 will see the new value as baseline вЂ” the `newValue = baseline === X ? Y : X` logic handles either starting state.
